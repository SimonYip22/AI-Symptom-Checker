# requirements.txt
# Lists all Python packages required to run the FastAPI API for the AI Symptom Checker

fastapi   # The web framework that lets you turn Python functions into a REST API
uvicorn   # ASGI server that runs the FastAPI app; handles incoming HTTP requests
pydantic  # Data validation and parsing library; ensures incoming requests have the correct structure and types

# Dev dependencies
pytest    # Testing framework for writing and running tests
httpx     # Required for TestClient in FastAPI/Starlette
